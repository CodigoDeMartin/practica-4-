procedure Poner_SemillasAl_(cantidad, dirección){
	/*
	Propósito: Poner una "cantidad" de semillas en la celda lindante en la dirección "dirección" dada
	Paramétro:
	 -cantidad tipo Número, es la cantidad de semillas que queremos poner.
	 -dirección tipo Dirección, es la dirección a la cual deceamos colocar la semilla.
	Precondición:
	 >no debe haber ninguna otra representacion que no sea de semillas en la celda actual.
	 >debe haber almenos una celda lindante en la dirección "dirección" dada
	*/
	Mover(dirección) 
	repeat(cantidad){
	Poner(Rojo)
	}
	Mover(opuesto(dirección)) //regresa el cabezal al inicio
}
