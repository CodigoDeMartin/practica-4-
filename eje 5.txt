procedure ProcesarCantidadDeLetrasDeLaPalabraYTranscribirAMayuscula(){

	Parametros: ninguno
	Precondicón:
	  >debe haber almenos una celda al norte por cada letra que haya en la palabra.
	*/

	PasarAlaSiguienteLetra()

	}
}
procedure PasarLetraActualAMayusculaYEscribirlaEnLaCeldaLindanteNorte(){
	/*
	Proposito: Pasar la letra de la celda actual a mayuscula y escribirla en la celda lindante norte
	Parametros: ninguno
	Precondicón:
	 debe haber almenos una letra en minuscula en la celda actual.
	 debe haber almenoss una celda lindante al norte de la posición actual del cabezal
	*/
	Mover(Norte)
	AgregarLetra_EnLaCeldaActual( letraDeLaceldaSurPeroEnMayuscula() )
	Mover(Sur) // regresar el cabezal
}

	*/

}

	 >debe heber almenos una letra en minuscula en la celda lindante al sur
